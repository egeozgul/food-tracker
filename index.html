const { onRequest, onCall, HttpsError } = require("firebase-functions/v2/https"); const { defineSecret } = require("firebase-functions/params"); const admin = require("firebase-admin"); admin.initializeApp(); const db = admin.firestore(); const WHOOP_CLIENT_ID = defineSecret("WHOOP_CLIENT_ID"); const WHOOP_CLIENT_SECRET = defineSecret("WHOOP_CLIENT_SECRET"); const TOKEN_URL = "https://api.prod.whoop.com/oauth/oauth2/token"; const CYCLE_URL = "https://api.prod.whoop.com/developer/v2/cycle"; // ================================================================== // IMPORTANT: after your first deploy, confirm this matches the real // URL Firebase gives you for whoopCallback, and register that SAME // URL as the Redirect URI in the Whoop Developer Dashboard. // ================================================================== const PROJECT_ID = "food-log-5f090"; const REDIRECT_URI = `https://us-central1-${PROJECT_ID}.cloudfunctions.net/whoopCallback`; const APP_URL = "https://tracker.egeozgul.com"; // ---- Step 2 of the flow: Whoop redirects here with a one-time code ---- exports.whoopCallback = onRequest( { secrets: [WHOOP_CLIENT_ID, WHOOP_CLIENT_SECRET], maxInstances: 5 }, async (req, res) => { const { code, state, error } = req.query; if (error) { return res.redirect(`${APP_URL}/?whoop=error`); } const uid = String(state || "").split(".")[0]; if (!code || !uid) { return res.status(400).send("Missing authorization code or state."); } try { const tokenRes = await fetch(TOKEN_URL, { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: new URLSearchParams({ grant_type: "authorization_code", code: String(code), redirect_uri: REDIRECT_URI, client_id: WHOOP_CLIENT_ID.value(), client_secret: WHOOP_CLIENT_SECRET.value(), }), }); const tokenData = await tokenRes.json(); if (!tokenRes.ok) { console.error("Whoop token exchange failed:", tokenData); return res.redirect(`${APP_URL}/?whoop=error`); } await db.collection("whoopTokens").doc(uid).set({ access_token: tokenData.access_token, refresh_token: tokenData.refresh_token, expires_at: Date.now() + tokenData.expires_in * 1000, }); return res.redirect(`${APP_URL}/?whoop=connected`); } catch (err) { console.error(err); return res.redirect(`${APP_URL}/?whoop=error`); } } ); // ---- Helper: get a valid access token, refreshing if needed ---- async function getValidAccessToken(uid) { const docRef = db.collection("whoopTokens").doc(uid); const snap = await docRef.get(); if (!snap.exists) { throw new HttpsError("failed-precondition", "Whoop is not connected for this account."); } let data = snap.data(); if (Date.now() > data.expires_at - 60000) { const tokenRes = await fetch(TOKEN_URL, { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: new URLSearchParams({ grant_type: "refresh_token", refresh_token: data.refresh_token, client_id: WHOOP_CLIENT_ID.value(), client_secret: WHOOP_CLIENT_SECRET.value(), scope: "offline", }), }); const tokenData = await tokenRes.json(); if (!tokenRes.ok) { console.error("Whoop token refresh failed:", tokenData); throw new HttpsError("internal", "Could not refresh Whoop access token."); } data = { access_token: tokenData.access_token, refresh_token: tokenData.refresh_token || data.refresh_token, expires_at: Date.now() + tokenData.expires_in * 1000, }; await docRef.set(data); } return data.access_token; } // ---- Helper: convert a Whoop record's start time to its local calendar date (YYYY-MM-DD) ---- function localDateFromRecord(record) { const start = new Date(record.start); const offsetStr = record.timezone_offset || "+00:00"; const sign = offsetStr[0] === "-" ? -1 : 1; const [oh, om] = offsetStr.slice(1).split(":").map(Number); const offsetMinutes = sign * (oh * 60 + (om || 0)); const localMs = start.getTime() + offsetMinutes * 60000; const local = new Date(localMs); const y = local.getUTCFullYear(); const m = String(local.getUTCMonth() + 1).padStart(2, "0"); const d = String(local.getUTCDate()).padStart(2, "0"); return `${y}-${m}-${d}`; } // ---- Called from the app: get calories burned for one date (YYYY-MM-DD) ---- exports.getWhoopCalories = onCall( { secrets: [WHOOP_CLIENT_ID, WHOOP_CLIENT_SECRET], maxInstances: 5 }, async (request) => { if (!request.auth) throw new HttpsError("unauthenticated", "Sign in required."); const dateStr = request.data && request.data.date; if (!dateStr) throw new HttpsError("invalid-argument", "A date is required."); const accessToken = await getValidAccessToken(request.auth.uid); // Widen the window by a day on each side so we don't miss a cycle that // starts late the night before (in UTC) but is still "today" locally. const start = new Date(`${dateStr}T00:00:00.000Z`); start.setUTCDate(start.getUTCDate() - 1); const end = new Date(`${dateStr}T23:59:59.999Z`); end.setUTCDate(end.getUTCDate() + 1); const url = `${CYCLE_URL}?start=${encodeURIComponent(start.toISOString())}&end=${encodeURIComponent(end.toISOString())}&limit=10`; const res = await fetch(url, { headers: { Authorization: `Bearer ${accessToken}` } }); const data = await res.json(); if (!res.ok) { console.error("Whoop cycle fetch failed:", data); throw new HttpsError("internal", "Could not fetch data from Whoop."); } const records = data.records || []; // Only use the single cycle that actually started on the requested local date — // summing every cycle that merely overlaps the day double-counts sleep-to-sleep cycles. const match = records.find( (r) => r.score_state === "SCORED" && localDateFromRecord(r) === dateStr ); if (!match || !match.score || typeof match.score.kilojoule !== "number") { return { kcal: null, cyclesFound: records.length }; } return { kcal: Math.round(match.score.kilojoule / 4.184), cyclesFound: records.length }; } ); // ---- Called from the app: disconnect Whoop ---- exports.whoopDisconnect = onCall({ maxInstances: 5 }, async (request) => { if (!request.auth) throw new HttpsError("unauthenticated", "Sign in required."); await db.collection("whoopTokens").doc(request.auth.uid).delete(); return { ok: true }; });
